local QuestSystem = {}
QuestSystem.__index = QuestSystem
 
function QuestSystem.new(data)
  local self = setmetatable({}, QuestSystem)
  self._quests = data.quests or {}
  self._active = {}
  return self
end
 
function QuestSystem:assign(id)
  local q = self._quests[id]
  if not q then return false end
  self._active[id] = { progress = 0 }
  return true
end
